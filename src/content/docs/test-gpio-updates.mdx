#include "splashkit.h"

int main()

    raspi_init();
    gpio_pin led_pin = PIN_11;
    raspi_set_mode(led_pin, GPIO_OUTPUT);
    
    // Blink LED
    raspi_write(led_pin, GPIO_HIGH);
    delay(500);
    raspi_write(led_pin, GPIO_LOW);
    
    raspi_cleanup();
    return 0;
}

    remote_raspi_set_mode(pi, led_pin, GPIO_OUTPUT);
